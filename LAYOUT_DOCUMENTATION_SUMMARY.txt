🎭 TAILWIND CLASS CHEATSHEET
═══════════════════════════════════════════════════════════════════════════════

SPACING:
  gap-1 = 4px,  gap-2 = 8px,  gap-3 = 12px,  gap-4 = 16px
  space-y-3 = 12px between children (vertical)
  space-y-4 = 16px between children (vertical)

BORDERS:
  border-white/10 = default,  border-white/5 = subtle

COLORS (Status Badges):
  pending:  border-white/10 bg-black/30
  kept:     border-emerald-500/30 bg-emerald-500/20 text-emerald-300
  trashed:  border-rose-500/30 bg-rose-500/20 text-rose-300

COLORS (Buttons):
  Keep:     border-emerald-500/20 bg-emerald-500/10
  Delete:   border-rose-500/20 bg-rose-500/10
  Select:   border-sky-400 bg-sky-500

ROUNDED CORNERS:
  rounded-lg = 8px
  rounded-xl = 12px
  rounded-2xl = 16px
  rounded-3xl = 24px
  rounded-[28px] = 28px (custom)
